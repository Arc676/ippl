122  Field recursive_preconditioner(Field& u, unsigned int level) {
123  mesh_type& mesh = u.get_mesh();
124  layout_type& layout = u.getLayout();
125  // Define etas if not defined yet
126  if (eta_m == nullptr) {
127  // Precompute the etas for later use
128  eta_m = new double[level_m + 1];
129  eta_m[0] = 2.0 / ((alpha_m + beta_m) * (1.0 + zeta_m));
130  if (level_m > 0) {
131  eta_m[1] =
132  2.0
133  / (1.0 + 2 * alpha_m * eta_m[0] - alpha_m * eta_m[0] * alpha_m * eta_m[0]);
134  }
135  for (unsigned int i = 2; i < level_m + 1; ++i) {
136  eta_m[i] = 2.0 / (1.0 + 2 * eta_m[i - 1] - eta_m[i - 1] * eta_m[i - 1]);
137  }
138  }
139 
140  Field res(mesh, layout);
141  // Base case
142  if (level == 0) {
143  res = eta_m[0] * u;
144  return res;
145  }
146  // Recursive case
147  Field PAPr(mesh, layout);
148  Field Pr(mesh, layout);
149 
150  Pr = recursive_preconditioner(u, level - 1);
151  PAPr = op_m(Pr);
152  PAPr = recursive_preconditioner(PAPr, level - 1);
153  res = eta_m[level] * (2.0 * Pr - PAPr);
154  return res;
155  }
156 
157  Field operator()(Field& u) override { return recursive_preconditioner(u, level_m); }
158 
159  protected:
160  OperatorF op_m; // Operator to be preconditioned
161  double alpha_m; // Smallest Eigenvalue
162  double beta_m; // Largest Eigenvalue
163  unsigned int level_m; // Number of recursive calls
164  double zeta_m; // smallest (alpha + beta) is multiplied by (1+zeta) to avoid clustering of
165  // Eigenvalues
166  double* eta_m = nullptr; // Size is determined at runtime
167  };

216  Field operator()(Field& r) override {
217  mesh_type& mesh = r.get_mesh();
218  layout_type& layout = r.getLayout();
219 
220  Field res(mesh, layout);
221  Field x(mesh, layout);
222  Field x_old(mesh, layout);
223  Field A(mesh, layout);
224  Field z(mesh, layout);
225 
226  // Precompute the coefficients if not done yet
227  if (rho_m == nullptr) {
228  // Start precomputing the coefficients
229  theta_m = (beta_m + alpha_m) / 2.0 * (1.0 + zeta_m);
230  delta_m = (beta_m - alpha_m) / 2.0;
231  sigma_m = theta_m / delta_m;
232 
233  rho_m = new double[degree_m + 1];
234  rho_m[0] = 1.0 / sigma_m;
235  for (unsigned int i = 1; i < degree_m + 1; ++i) {
236  rho_m[i] = 1.0 / (2.0 * sigma_m - rho_m[i - 1]);
237  }
238  } // End of precomputing the coefficients
239 
240  res = r.deepCopy();
241 
242  x_old = r / theta_m;
243  A = op_m(r);
244  x = 2.0 * rho_m[1] / delta_m * (2.0 * r - A / theta_m);
245 
246  if (degree_m == 0) {
247  return x_old;
248  }
249 
250  if (degree_m == 1) {
251  return x;
252  }
253  for (unsigned int i = 2; i < degree_m + 1; ++i) {
254  A = op_m(x);
255  z = 2.0 / delta_m * (r - A);
256  res = rho_m[i] * (2 * sigma_m * x - rho_m[i - 1] * x_old + z);
257  x_old = x.deepCopy();
258  x = res.deepCopy();
259  }
260  return res;
261  }
262

268  double theta_m;
269  double sigma_m;

376  double powermethod(Functor&& f, Field& x_0, unsigned int max_iter = 5000, double tol = 1e-3) {
377  unsigned int i = 0;
378  using mesh_type = typename Field::Mesh_t;
379  using layout_type = typename Field::Layout_t;
380  mesh_type& mesh = x_0.get_mesh();
381  layout_type& layout = x_0.getLayout();
382  Field x_new(mesh, layout);
383  Field x_diff(mesh, layout);
384  double error = 1.0;
385  double lambda = 1.0;
386  while (error > tol && i < max_iter) {
387  x_new = f(x_0);
388  lambda = norm(x_new);
389  x_diff = x_new - lambda * x_0;
390  error = norm(x_diff);
391  x_new = x_new / lambda;
392  x_0 = x_new.deepCopy();
393  ++i;
394  }
395  if (i == max_iter) {
396  std::cerr << "Powermethod did not converge, lambda_max : " << lambda
397  << ", error : " << error << std::endl;
398  }
399  return lambda;
